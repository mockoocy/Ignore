<function name=fibonacci n:int returnType=int > 
    <if condition={ n < 0 }>
        <throw exceptionType=InvalidArgumentExeption >
                "There are no negative fib numbers"
        </throw>
    </if>
    <if condition={n == 1 || n == 0}> 
        <return>
            n
        </return>
    </if>
    <return>
        fibonacci(n - 1) + fibonacci(n - 2)
    </return>
</function>

<program>
    <io>
        print(fibonacci(2+2*3));
    </io>
</program>
